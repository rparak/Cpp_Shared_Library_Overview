def Max(x):
    """
    Description:
        Function to return the maximum number from the list.
            
    Args:
        (1) x [Float Array]: Input list for determining the value.
    Returns:
        (1) parameter [Float]: The maximum value of the list.
        (2) parameter [INT]: Index of the maximum value in the list.
    """

    max_x = {'Value': x[0], 'Index': 0}

    for i, x_i in enumerate(x[1:]):
        if x_i > max_x['Value']:
            max_x['Value'] = x_i
            max_x['Index'] = i + 1

    return max_x
